    function strMapToObj(strMap) {
    let obj = Object.create(null);
    for (let [k,v] of strMap) {
    obj[k] = v;
    }
    return obj;
    }

    const myMap = new Map()
    .set(‘yes’, true)
    .set(‘no’, false);
    strMapToObj(myMap)
    // { yes: true, no: false }

    function objToStrMap(obj) {
    let strMap = new Map();
    for (let k of Object.keys(obj)) {
    strMap.set(k, obj[k]);
    }
    return strMap;
    }
